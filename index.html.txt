function abrirPresente(){document.getElementById('entrada').style.opacity='0';setTimeout(()=>{document.getElementById('entrada').style.display='none';document.getElementById('conteudo').classList.add('ativo');document.getElementById('musica').play();},800);}
function atualizarContador(){
const inicio=new Date('2025-09-01');
const agora=new Date();
let diff=agora-inicio;
const dias=Math.floor(diff/86400000);
diff-=dias*86400000;
const horas=Math.floor(diff/3600000);
diff-=horas*3600000;
const min=Math.floor(diff/60000);
diff-=min*60000;
const seg=Math.floor(diff/1000);
const horasTotais=Math.floor((agora-inicio)/3600000);
document.getElementById('contador').innerHTML=`Juntos há ${dias} dias, ${horas} horas, ${min} minutos e ${seg} segundos<br><span style="font-size:18px;color:#555;margin-top:8px;display:block;">Isso são ${horasTotais} horas de amor ❤️</span>`;
setTimeout(atualizarContador,1000);
}
atualizarContador();
